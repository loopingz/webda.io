# @webda/debug API Backend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create a `@webda/debug` package with a debug server (port 18181) that exposes REST introspection endpoints and a WebSocket for live request logging, plus a `webda debug` CLI command.

**Architecture:** New `packages/debug` package with a `DebugService` that starts a separate HTTP+WS server alongside the app server. Pure introspection functions read from existing `@webda/core` hooks. A ring buffer captures request events and pushes them to WebSocket clients. The debug server survives Core restarts in watch mode.

**Tech Stack:** TypeScript, `@webda/core`, `ws` (WebSocket server), `node:http`

```typescript
/**
 * A single entry in the request log
 */
export interface RequestLogEntry {
  /** Unique request identifier */
  id: string;
  /** HTTP method */
  method: string;
  /** Request URL */
  url: string;
  /** Unix timestamp (ms) when the request started */
  timestamp: number;
  /** HTTP response status code (set on completion) */
  statusCode?: number;
  /** Request duration in ms (set on completion) */
  duration?: number;
}

/**
 * Event pushed to subscribers
 */
export type RequestLogEvent =
  | { type: "request"; id: string; method: string; url: string; timestamp: number }
  | { type: "result"; id: string; statusCode: number; duration: number }
  | { type: "404"; id: string; method: string; url: string };

type EventCallback = (event: RequestLogEvent) => void;

/**
 * Fixed-size ring buffer for tracking HTTP requests.
 * Stores recent entries and pushes events to subscribers.
 */
export class RequestLog {
  private entries: RequestLogEntry[] = [];
  private index = new Map<string, RequestLogEntry>();
  private subscribers: Set<EventCallback> = new Set();

  /**
   * Create a new RequestLog
   * @param maxSize - maximum number of entries to retain
   */
  constructor(private maxSize: number = 1000) {}

  /**
   * Record the start of a request
   * @param id - unique request identifier
   * @param method - HTTP method
   * @param url - request URL
   */
  startRequest(id: string, method: string, url: string): void {
    const entry: RequestLogEntry = { id, method, url, timestamp: Date.now() };
    if (this.entries.length >= this.maxSize) {
      const removed = this.entries.shift();
      if (removed) this.index.delete(removed.id);
    }
    this.entries.push(entry);
    this.index.set(id, entry);
    this.notify({ type: "request", id, method, url, timestamp: entry.timestamp });
  }

  /**
   * Complete a request with status and duration
   * @param id - request identifier
   * @param statusCode - HTTP status code
   * @param duration - request duration in ms
   */
  completeRequest(id: string, statusCode: number, duration: number): void {
    const entry = this.index.get(id);
    if (!entry) return;
    entry.statusCode = statusCode;
    entry.duration = duration;
    this.notify({ type: "result", id, statusCode, duration });
  }

  /**
   * Mark a request as 404
   * @param id - request identifier
   */
  markNotFound(id: string): void {
    const entry = this.index.get(id);
    if (!entry) return;
    entry.statusCode = 404;
    this.notify({ type: "404", id, method: entry.method, url: entry.url });
  }

  /**
   * Get all entries (oldest first)
   * @returns copy of current entries
   */
  getEntries(): RequestLogEntry[] {
    return [...this.entries];
  }

  /**
   * Subscribe to events
   * @param callback - called on each event
   * @returns unsubscribe function
   */
  onEvent(callback: EventCallback): () => void {
    this.subscribers.add(callback);
    return () => this.subscribers.delete(callback);
  }

  private notify(event: RequestLogEvent): void {
    for (const cb of this.subscribers) {
      cb(event);
    }
  }
}
```

```typescript
import { useApplication, useCore, useRouter, listOperations, useModelMetadata } from "@webda/core";

/**
 * Model info returned by the introspection API
 */
export interface ModelInfo {
  id: string;
  plural: string;
  actions: string[];
  relations: any;
  metadata: any;
}

/**
 * Service info returned by the introspection API
 */
export interface ServiceInfo {
  name: string;
  type: string;
  state: string;
  capabilities: Record<string, any>;
}

/**
 * Operation info returned by the introspection API
 */
export interface OperationInfo {
  id: string;
  input?: string;
  output?: string;
  parameters?: string;
  [key: string]: any;
}

/**
 * Route info returned by the introspection API
 */
export interface RouteInfoEntry {
  path: string;
  methods: string[];
  executor: string;
}

/**
 * Get all models with their metadata
 * @returns array of model information
 */
export function getModels(): ModelInfo[] {
  const app = useApplication();
  const models = app.getModels();
  return Object.entries(models).map(([key, model]) => {
    const metadata = useModelMetadata(model);
    return {
      id: metadata?.Identifier || key,
      plural: metadata?.Plural || key,
      actions: Object.keys(metadata?.Actions || {}),
      relations: metadata?.Relations || {},
      metadata
    };
  });
}

/**
 * Get a single model by identifier
 * @param id - model identifier (e.g. "MyApp/Task")
 * @returns model info or undefined
 */
export function getModel(id: string): ModelInfo | undefined {
  return getModels().find(m => m.id === id);
}

/**
 * Get all services with state and configuration
 * @returns array of service information
 */
export function getServices(): ServiceInfo[] {
  const core = useCore();
  const services = core.getServices();
  return Object.entries(services)
    .filter(([, svc]) => svc != null)
    .map(([name, svc]) => ({
      name,
      type: svc.getParameters()?.type || "unknown",
      state: svc.getState(),
      capabilities: svc.getCapabilities()
    }));
}

/**
 * Get all registered operations
 * @returns array of operation information
 */
export function getOperations(): OperationInfo[] {
  const ops = listOperations();
  return Object.entries(ops).map(([id, def]) => ({
    id,
    ...def
  }));
}

/**
 * Get all registered routes
 * @returns array of route information
 */
export function getRoutes(): RouteInfoEntry[] {
  const router = useRouter();
  const routes = router.getRoutes();
  const result: RouteInfoEntry[] = [];
  for (const [path, infos] of Object.entries(routes)) {
    for (const info of infos as any[]) {
      result.push({
        path,
        methods: info.methods,
        executor: info.executor
      });
    }
  }
  return result;
}

/**
 * Get the resolved application configuration
 * @returns configuration object
 */
export function getConfig(): Record<string, any> {
  return useApplication().getConfiguration();
}
```

```typescript
import { Service, ServiceParameters, useDynamicService, useRouter } from "@webda/core";
import { Command } from "@webda/core/services/command";
import { useCoreEvents } from "@webda/core/events/events";
import { useLog } from "@webda/workout";
import { createServer, IncomingMessage, ServerResponse } from "node:http";
import type { Server } from "node:http";
import { WebSocketServer, WebSocket } from "ws";
import { RequestLog, type RequestLogEvent } from "./requestlog.js";
import { getModels, getModel, getServices, getOperations, getRoutes, getConfig } from "./introspection.js";

/**
 * Parameters for DebugService
 */
class DebugServiceParameters extends ServiceParameters {}

/**
 * Debug server providing introspection REST API and WebSocket live events.
 * Runs on a separate port from the application server.
 *
 * @WebdaModda
 */
export class DebugService extends Service<DebugServiceParameters> {
  private server?: Server;
  private wss?: WebSocketServer;
  private requestLog: RequestLog = new RequestLog();
  private unsubscribeEvents: (() => void)[] = [];

  /**
   * Subscribe to core events for request logging
   * @returns this for chaining
   */
  resolve() {
    this.subscribeToEvents();
    return super.resolve();
  }

  /**
   * Subscribe to Webda core events and forward to the request log
   */
  subscribeToEvents(): void {
    this.unsubscribeFromEvents();
    const requestTimers = new Map<string, number>();

    this.unsubscribeEvents.push(
      useCoreEvents("Webda.Request", ({ context }) => {
        const http = context.getHttpContext();
        const id = Math.random().toString(36).substring(2);
        context.setExtension("debugRequestId", id);
        requestTimers.set(id, Date.now());
        this.requestLog.startRequest(id, http.getMethod(), http.getUrl());
      })
    );

    this.unsubscribeEvents.push(
      useCoreEvents("Webda.Result", ({ context }) => {
        const id = context.getExtension<string>("debugRequestId");
        if (!id) return;
        const start = requestTimers.get(id);
        const duration = start ? Date.now() - start : 0;
        requestTimers.delete(id);
        this.requestLog.completeRequest(id, context.statusCode || 200, duration);
      })
    );

    this.unsubscribeEvents.push(
      useCoreEvents("Webda.404", ({ context }) => {
        const id = context.getExtension<string>("debugRequestId");
        if (!id) return;
        this.requestLog.markNotFound(id);
      })
    );
  }

  /**
   * Remove all event subscriptions
   */
  unsubscribeFromEvents(): void {
    for (const unsub of this.unsubscribeEvents) {
      unsub();
    }
    this.unsubscribeEvents = [];
  }

  /**
   * Start the debug server with introspection API and WebSocket
   * @param port - debug server port
   * @param servePort - application server port (passed to HttpServer.serve)
   */
  @Command("debug", {
    description: "Start dev server with debug dashboard",
    requires: ["router", "rest-domain"]
  })
  async debug(
    /** @alias p @description Debug server port */
    port: number = 18181,
    /** @description Application server port */
    servePort: number = 18080
  ) {
    // Start the application server
    const httpServer = useDynamicService<any>("HttpServer");
    if (httpServer?.serve) {
      await httpServer.serve(undefined, servePort);
    }

    // Start the debug server
    this.server = createServer((req, res) => this.handleRequest(req, res));
    this.wss = new WebSocketServer({ server: this.server });

    // Push request log events to WebSocket clients
    this.requestLog.onEvent(event => {
      this.broadcast(event);
    });

    this.server.listen(port, () => {
      useLog("INFO", `Debug server listening on http://localhost:${port}`);
    });
  }

  /**
   * Route incoming HTTP requests to the appropriate handler
   * @param req - the incoming request
   * @param res - the server response
   */
  private handleRequest(req: IncomingMessage, res: ServerResponse): void {
    // CORS headers for dev
    res.setHeader("Access-Control-Allow-Origin", "*");
    res.setHeader("Access-Control-Allow-Methods", "GET, OPTIONS");
    res.setHeader("Access-Control-Allow-Headers", "Content-Type");

    if (req.method === "OPTIONS") {
      res.writeHead(204);
      res.end();
      return;
    }

    const url = req.url || "/";

    try {
      let body: any;
      if (url === "/api/models") {
        body = getModels();
      } else if (url.startsWith("/api/models/")) {
        const id = decodeURIComponent(url.substring("/api/models/".length));
        body = getModel(id);
        if (!body) {
          res.writeHead(404);
          res.end(JSON.stringify({ error: "Model not found" }));
          return;
        }
      } else if (url === "/api/services") {
        body = getServices();
      } else if (url === "/api/operations") {
        body = getOperations();
      } else if (url === "/api/routes") {
        body = getRoutes();
      } else if (url === "/api/config") {
        body = getConfig();
      } else if (url === "/api/openapi") {
        body = useRouter().exportOpenAPI(false);
      } else if (url === "/api/requests") {
        body = this.requestLog.getEntries();
      } else {
        res.writeHead(404);
        res.end(JSON.stringify({ error: "Not found" }));
        return;
      }

      res.writeHead(200, { "Content-Type": "application/json" });
      res.end(JSON.stringify(body, undefined, 2));
    } catch (err) {
      res.writeHead(500);
      res.end(JSON.stringify({ error: err.message }));
    }
  }

  /**
   * Broadcast a message to all connected WebSocket clients
   * @param data - the data to send
   */
  broadcast(data: RequestLogEvent | { type: "restart" }): void {
    if (!this.wss) return;
    const msg = JSON.stringify(data);
    for (const client of this.wss.clients) {
      if (client.readyState === WebSocket.OPEN) {
        client.send(msg);
      }
    }
  }

  /**
   * Stop the debug server
   */
  async stop(): Promise<void> {
    this.unsubscribeFromEvents();
    if (this.wss) {
      for (const client of this.wss.clients) {
        client.close();
      }
      this.wss.close();
      this.wss = undefined;
    }
    if (this.server) {
      await new Promise<void>(resolve => this.server!.close(() => resolve()));
      this.server = undefined;
    }
    await super.stop();
  }
}
```
